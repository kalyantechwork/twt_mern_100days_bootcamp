1. What is Mongoose?

Mongoose is an Object Data Modeling (ODM) library for MongoDB and Node.js. It provides a schema-based solution to model your application data with built-in validation, query building, and middleware.

Without Mongoose

• Raw MongoDB driver
• Manual validation
• No schema enforcement
• Manual type conversion
• More boilerplate code

db.collection('users').insertOne({ name: "Ramesh", age: "25" // string, not number! })

With Mongoose

• Schema validation
• Built-in data types
• Automatic type casting
• Middleware (pre/post hooks)
• Query helpers

const User = mongoose.model('User', userSchema); await User.create({ name: "Ramesh", age: "25" }); // age automatically becomes 25 (number)
Mongoose sits between your Node.js app and MongoDB:
Node.js App → Mongoose (ODM) → MongoDB Driver → MongoDB Database
2. Why Use Mongoose in MERN Stack?
📐 Schema Definition

Define the structure of your documents with data types, validation, and defaults

✅ Built-in Validation

Required fields, min/max length, email format, custom validators

🔗 Relationship Management

Easily define references between collections (populate)

🪝 Middleware (Hooks)

pre and post hooks for operations (e.g., hash password before save)

📝 Query Builder

Chainable queries with promises (find, where, limit, populate)

⚡ Performance

Built-in connection pooling, indexing support

4. Connecting to MongoDB with Mongoose
📁 Connection String Formats
// Local MongoDB mongodb://localhost:27017/database_name // MongoDB Atlas (Cloud) mongodb+srv://username:password@cluster.mongodb.net/database_name // With options mongodb://localhost:27017/myapp?retryWrites=true&w=majority
📁 .env file (Environment Variables)
# .env PORT=5000 MONGODB_URI=mongodb://localhost:27017/mongooseDemo
📁 config/db.js – Database Connection Setup
// config/db.js const mongoose = require('mongoose'); const connectDB = async () => { try { const conn = await mongoose.connect(process.env.MONGODB_URI, { // These options are no longer needed in mongoose 6+, but kept for reference useNewUrlParser: true, useUnifiedTopology: true, }); console.log(`✅ MongoDB Connected: ${conn.connection.host}`); } catch (error) { console.error(`❌ Error: ${error.message}`); process.exit(1); } }; module.exports = connectDB;
📁 server.js – Main File
// server.js require('dotenv').config(); const mongoose = require('mongoose'); const connectDB = require('./config/db'); // Connect to database connectDB(); // Listen for mongoose events mongoose.connection.on('connected', () => { console.log('🔌 Mongoose connected to MongoDB'); }); mongoose.connection.on('error', (err) => { console.log(`❌ Mongoose connection error: ${err}`); }); mongoose.connection.on('disconnected', () => { console.log('📴 Mongoose disconnected'); }); // Handle application termination process.on('SIGINT', async () => { await mongoose.connection.close(); console.log('Mongoose connection closed due to app termination'); process.exit(0); }); const PORT = process.env.PORT || 5000; const app = require('./app'); // Your Express app app.listen(PORT, () => { console.log(`🚀 Server running on port ${PORT}`); });
Connection Success! You should see: "✅ MongoDB Connected: localhost" or Atlas hostname
5. Defining a Schema and Model
// models/User.js const mongoose = require('mongoose'); // Define the schema (structure of document) const userSchema = new mongoose.Schema({ name: { type: String, required: [true, 'Name is required'], trim: true, minlength: [2, 'Name must be at least 2 characters'], maxlength: [50, 'Name cannot exceed 50 characters'] }, email: { type: String, required: [true, 'Email is required'], unique: true, lowercase: true, trim: true, match: [/^\S+@\S+\.\S+$/, 'Please enter a valid email'] }, age: { type: Number, min: [0, 'Age cannot be negative'], max: [120, 'Age cannot exceed 120'] }, password: { type: String, required: [true, 'Password is required'], minlength: [6, 'Password must be at least 6 characters'], select: false // Don't return password by default }, role: { type: String, enum: ['user', 'admin', 'moderator'], default: 'user' }, isActive: { type: Boolean, default: true }, hobbies: { type: [String], default: [] }, address: { street: String, city: String, zipCode: String }, createdAt: { type: Date, default: Date.now, immutable: true // Cannot be changed after creation }, updatedAt: { type: Date, default: Date.now } }, { timestamps: true, // Automatically adds createdAt and updatedAt toJSON: { virtuals: true }, toObject: { virtuals: true } }); // Virtual field (not stored in DB) userSchema.virtual('fullInfo').get(function() { return `${this.name} (${this.email})`; }); // Middleware: Runs before saving userSchema.pre('save', function(next) { this.updatedAt = Date.now(); console.log(`🔄 Saving user: ${this.name}`); next(); }); // Middleware: Runs after saving userSchema.post('save', function(doc, next) { console.log(`✅ User saved: ${doc.name}`); next(); }); // Instance method userSchema.methods.greet = function() { return `Hello, my name is ${this.name}`; }; // Static method userSchema.statics.findByEmail = function(email) { return this.findOne({ email }); }; // Create the model (singular name, mongoose creates plural collection) const User = mongoose.model('User', userSchema); module.exports = User;
6. CRUD Operations with Mongoose
📝 CREATE – Saving Documents
// Method 1: Create and save const newUser = new User({ name: "Ramesh", email: "ramesh@example.com", age: 25, password: "password123", hobbies: ["coding", "reading"] }); const savedUser = await newUser.save(); // Method 2: Create directly const user = await User.create({ name: "Priya", email: "priya@example.com", age: 22, password: "password456", hobbies: ["dancing", "music"] }); // Method 3: Insert many const users = await User.insertMany([ { name: "Suresh", email: "suresh@example.com", password: "pass789" }, { name: "Anjali", email: "anjali@example.com", password: "pass101" } ]);
📖 READ – Finding Documents
// Find all users const allUsers = await User.find(); // Find with filter const activeUsers = await User.find({ isActive: true }); // Find one by ID const user = await User.findById("507f1f77bcf86cd799439011"); // Find one by filter const userByEmail = await User.findOne({ email: "ramesh@example.com" }); // Count documents const count = await User.countDocuments({ isActive: true }); // Query with multiple conditions const users = await User.find({ age: { $gte: 18, $lte: 60 }, isActive: true }).select('name email age').sort({ age: -1 }).limit(5); // Select specific fields (excluding password) const usersWithoutPassword = await User.find().select('-password'); // Populate (for references – we'll cover later) // const posts = await Post.find().populate('author');
✏️ UPDATE – Modifying Documents
// Update one const updated = await User.updateOne( { email: "ramesh@example.com" }, { $set: { age: 26 }, $inc: { version: 1 } } ); // Update many await User.updateMany( { isActive: true }, { $set: { membership: "premium" } } ); // Find by ID and update const user = await User.findByIdAndUpdate( "507f1f77bcf86cd799439011", { age: 27 }, { new: true, runValidators: true } // Return updated doc, run validation ); // Find one and update const updatedUser = await User.findOneAndUpdate( { email: "priya@example.com" }, { $push: { hobbies: "singing" } }, { new: true } );
🗑️ DELETE – Removing Documents
// Delete one await User.deleteOne({ email: "suresh@example.com" }); // Delete many await User.deleteMany({ isActive: false }); // Find by ID and delete const deletedUser = await User.findByIdAndDelete("507f1f77bcf86cd799439011"); // Find one and delete const deleted = await User.findOneAndDelete({ email: "anjali@example.com" });
7. Complete Working Example
// app.js – Complete Express + Mongoose Example require('dotenv').config(); const express = require('express'); const mongoose = require('mongoose'); const app = express(); app.use(express.json()); // Connect to MongoDB mongoose.connect(process.env.MONGODB_URI) .then(() => console.log('✅ Connected to MongoDB')) .catch(err => console.error('Connection error:', err)); // User Schema const userSchema = new mongoose.Schema({ name: { type: String, required: true }, email: { type: String, required: true, unique: true }, age: { type: Number, min: 0, max: 120 } }, { timestamps: true }); const User = mongoose.model('User', userSchema); // ========== API ROUTES ========== // CREATE – POST /api/users app.post('/api/users', async (req, res) => { try { const user = await User.create(req.body); res.status(201).json({ success: true, data: user }); } catch (error) { res.status(400).json({ success: false, message: error.message }); } }); // READ ALL – GET /api/users app.get('/api/users', async (req, res) => { try { const users = await User.find().sort({ createdAt: -1 }); res.json({ success: true, count: users.length, data: users }); } catch (error) { res.status(500).json({ success: false, message: error.message }); } }); // READ ONE – GET /api/users/:id app.get('/api/users/:id', async (req, res) => { try { const user = await User.findById(req.params.id); if (!user) { return res.status(404).json({ success: false, message: 'User not found' }); } res.json({ success: true, data: user }); } catch (error) { res.status(500).json({ success: false, message: error.message }); } }); // UPDATE – PUT /api/users/:id app.put('/api/users/:id', async (req, res) => { try { const user = await User.findByIdAndUpdate( req.params.id, req.body, { new: true, runValidators: true } ); if (!user) { return res.status(404).json({ success: false, message: 'User not found' }); } res.json({ success: true, data: user }); } catch (error) { res.status(400).json({ success: false, message: error.message }); } }); // DELETE – DELETE /api/users/:id app.delete('/api/users/:id', async (req, res) => { try { const user = await User.findByIdAndDelete(req.params.id); if (!user) { return res.status(404).json({ success: false, message: 'User not found' }); } res.json({ success: true, message: 'User deleted successfully' }); } catch (error) { res.status(500).json({ success: false, message: error.message }); } }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => { console.log(`🚀 Server running on port ${PORT}`); });
