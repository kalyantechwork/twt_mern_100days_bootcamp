1. Why Pagination & Filtering Matter
Without Pagination:
• Loading 10,000 products at once = Slow app + Bad UX
• No way to search or filter = Users can't find what they want
• Database overload = Expensive and slow

With Pagination & Filtering:
• Fast responses (only load what's needed)
• Users can search, sort, and filter
• Better performance and lower costs

4. Advanced Pagination with Metadata
// Advanced pagination with more metadata class APIFeatures { constructor(query, queryString) { this.query = query; this.queryString = queryString; } paginate() { const page = parseInt(this.queryString.page) || 1; const limit = parseInt(this.queryString.limit) || 10; const skip = (page - 1) * limit; this.query = this.query.skip(skip).limit(limit); this.paginationResult = { currentPage: page, limit: limit, skip: skip }; return this; } async getPaginationInfo(total) { const { currentPage, limit } = this.paginationResult; const totalPages = Math.ceil(total / limit); return { currentPage, totalPages, totalItems: total, itemsPerPage: limit, hasNextPage: currentPage < totalPages, hasPrevPage: currentPage > 1, nextPage: currentPage < totalPages ? currentPage + 1 : null, prevPage: currentPage > 1 ? currentPage - 1 : null }; } } // Using cursor-based pagination (better for infinite scroll) // Uses last seen ID instead of skip (more efficient for large datasets) const cursorBasedPagination = async (req, res) => { const limit = parseInt(req.query.limit) || 10; const lastId = req.query.cursor; // Last document ID from previous page let query = {}; if (lastId) { query._id = { $gt: lastId }; } const products = await Product.find(query) .limit(limit + 1) // Get one extra to check if more exist .sort({ _id: 1 }); const hasNextPage = products.length > limit; const items = hasNextPage ? products.slice(0, -1) : products; const nextCursor = hasNextPage ? items[items.length - 1]._id : null; res.json({ success: true, data: items, pagination: { hasNextPage, nextCursor, count: items.length } }); };
5. Basic Filtering
// GET /api/products?category=Electronics&brand=Apple const filter = {}; if (req.query.category) { filter.category = req.query.category; } if (req.query.brand) { filter.brand = req.query.brand; } if (req.query.minPrice) { filter.price = { $gte: parseFloat(req.query.minPrice) }; } if (req.query.maxPrice) { filter.price = { ...filter.price, $lte: parseFloat(req.query.maxPrice) }; } if (req.query.rating) { filter.rating = { $gte: parseFloat(req.query.rating) }; } if (req.query.inStock === 'true') { filter.stock = { $gt: 0 }; } const products = await Product.find(filter); // Example URLs: // /api/products?category=Electronics&minPrice=50&maxPrice=500 // /api/products?brand=Apple&rating=4.5 // /api/products?inStock=true
6. Advanced Filtering (Dynamic Query Builder)
class AdvancedFilter { constructor(queryParams) { this.filter = {}; this.queryParams = queryParams; } // String filters (exact match) addStringFilter(field) { if (this.queryParams[field]) { this.filter[field] = this.queryParams[field]; } return this; } // Number range filter addNumberRangeFilter(field, minKey, maxKey) { if (this.queryParams[minKey] || this.queryParams[maxKey]) { this.filter[field] = {}; if (this.queryParams[minKey]) { this.filter[field].$gte = parseFloat(this.queryParams[minKey]); } if (this.queryParams[maxKey]) { this.filter[field].$lte = parseFloat(this.queryParams[maxKey]); } } return this; } // Multiple values filter (IN operator) addMultiValueFilter(field) { if (this.queryParams[field]) { const values = this.queryParams[field].split(','); this.filter[field] = { $in: values }; } return this; } // Array contains filter addArrayContainsFilter(field) { if (this.queryParams[field]) { this.filter[field] = this.queryParams[field]; } return this; } // Text search (case-insensitive) addTextSearch(fields) { if (this.queryParams.search) { const searchRegex = new RegExp(this.queryParams.search, 'i'); this.filter.$or = fields.map(field => ({ [field]: searchRegex })); } return this; } // Date range filter addDateRangeFilter(field, fromKey, toKey) { if (this.queryParams[fromKey] || this.queryParams[toKey]) { this.filter[field] = {}; if (this.queryParams[fromKey]) { this.filter[field].$gte = new Date(this.queryParams[fromKey]); } if (this.queryParams[toKey]) { this.filter[field].$lte = new Date(this.queryParams[toKey]); } } return this; } build() { return this.filter; } } // Usage in route app.get('/api/products', async (req, res) => { const filter = new AdvancedFilter(req.query) .addStringFilter('category') .addStringFilter('brand') .addNumberRangeFilter('price', 'minPrice', 'maxPrice') .addNumberRangeFilter('rating', 'minRating', null) .addMultiValueFilter('tags') .addTextSearch(['name', 'description', 'brand']) .addDateRangeFilter('createdAt', 'fromDate', 'toDate') .build(); const products = await Product.find(filter); res.json({ success: true, count: products.length, filters: req.query, data: products }); }); // Example URL with multiple filters: // /api/products?category=Electronics&minPrice=50&maxPrice=500&tags=gaming,accessory&search=laptop&fromDate=2024-01-01

8. Text Search (Full-Text Search)
// First, create a text index (run once) await Product.createIndexes([ { name: "text", description: "text", tags: "text" } ]); // Or in schema definition: productSchema.index({ name: 'text', description: 'text', tags: 'text' }); // Basic text search const searchTerm = req.query.search; const products = await Product.find({ $text: { $search: searchTerm } }); // Text search with sorting by relevance const products = await Product.find( { $text: { $search: searchTerm } }, { score: { $meta: "textScore" } } ).sort({ score: { $meta: "textScore" } }); // Text search with language support const products = await Product.find({ $text: { $search: searchTerm, $language: "en", $caseSensitive: false, $diacriticSensitive: false } }); // Text search with excluded words (using -) // Search for "laptop" but NOT "gaming" const products = await Product.find({ $text: { $search: "laptop -gaming" } }); // Text search phrase (exact match in quotes) const products = await Product.find({ $text: { $search: "\"mechanical keyboard\"" } }); // Combining text search with other filters const products = await Product.find({ $text: { $search: searchTerm }, category: "Electronics", price: { $gte: 100 } });
9. Complete APIFeatures Class (Production Ready)
// utils/apiFeatures.js class APIFeatures { constructor(query, queryString) { this.query = query; this.queryString = queryString; } // 1. Filtering filter() { const queryObj = { ...this.queryString }; const excludedFields = ['page', 'sort', 'limit', 'fields', 'search']; excludedFields.forEach(el => delete queryObj[el]); // Advanced filtering (gt, gte, lt, lte, in) let queryStr = JSON.stringify(queryObj); queryStr = queryStr.replace(/\b(gt|gte|lt|lte|in)\b/g, match => `$${match}`); this.query = this.query.find(JSON.parse(queryStr)); return this; } // 2. Text Search search() { if (this.queryString.search) { this.query = this.query.find({ $text: { $search: this.queryString.search } }); } return this; } // 3. Sorting sort() { if (this.queryString.sort) { const sortBy = this.queryString.sort.split(',').join(' '); this.query = this.query.sort(sortBy); } else { this.query = this.query.sort('-createdAt'); } return this; } // 4. Field Limiting limitFields() { if (this.queryString.fields) { const fields = this.queryString.fields.split(',').join(' '); this.query = this.query.select(fields); } else { this.query = this.query.select('-__v'); } return this; } // 5. Pagination paginate() { const page = parseInt(this.queryString.page) || 1; const limit = parseInt(this.queryString.limit) || 10; const skip = (page - 1) * limit; this.query = this.query.skip(skip).limit(limit); this.paginationInfo = { page, limit, skip }; return this; } // Get pagination metadata async getPaginationMetadata(totalDocuments) { const { page, limit } = this.paginationInfo; const totalPages = Math.ceil(totalDocuments / limit); return { currentPage: page, totalPages, totalDocuments, documentsPerPage: limit, hasNextPage: page < totalPages, hasPrevPage: page > 1, nextPage: page < totalPages ? page + 1 : null, prevPage: page > 1 ? page - 1 : null }; } } // Usage in route app.get('/api/products', async (req, res) => { try { // Build query with all features const features = new APIFeatures(Product.find(), req.query) .filter() // Filter by fields .search() // Text search .sort() // Sort results .limitFields() // Select specific fields .paginate(); // Paginate // Execute query const products = await features.query; // Get total count for pagination (without pagination) const totalQuery = new APIFeatures(Product.find(), req.query) .filter() .search(); const total = await totalQuery.query.countDocuments(); const pagination = await features.getPaginationMetadata(total); res.json({ success: true, results: products.length, pagination, data: products }); } catch (error) { res.status(400).json({ success: false, message: error.message }); } }); // Example URL: // /api/products?category=Electronics&price[gte]=50&price[lte]=500&sort=price,rating&fields=name,price,rating&page=2&limit=10&search=laptop
10. Aggregation Pipeline for Advanced Filtering
// Aggregation pipeline for complex queries const getProductsWithStats = async (req, res) => { const { category, minPrice, maxPrice, sortBy, limit } = req.query; const pipeline = []; // Stage 1: Match (filter) const matchStage = {}; if (category) matchStage.category = category; if (minPrice || maxPrice) { matchStage.price = {}; if (minPrice) matchStage.price.$gte = parseFloat(minPrice); if (maxPrice) matchStage.price.$lte = parseFloat(maxPrice); } if (Object.keys(matchStage).length > 0) { pipeline.push({ $match: matchStage }); } // Stage 2: Lookup (join with reviews) pipeline.push({ $lookup: { from: 'reviews', localField: '_id', foreignField: 'productId', as: 'reviews' } }); // Stage 3: Add fields (calculate average rating) pipeline.push({ $addFields: { averageRating: { $avg: '$reviews.rating' }, reviewCount: { $size: '$reviews' } } }); // Stage 4: Sort if (sortBy) { const sortOrder = req.query.sortOrder === 'desc' ? -1 : 1; pipeline.push({ $sort: { [sortBy]: sortOrder } }); } // Stage 5: Pagination const page = parseInt(req.query.page) || 1; const limitNum = parseInt(limit) || 10; const skip = (page - 1) * limitNum; pipeline.push({ $skip: skip }); pipeline.push({ $limit: limitNum }); // Stage 6: Project (select fields) pipeline.push({ $project: { name: 1, price: 1, category: 1, averageRating: 1, reviewCount: 1, stock: 1 } }); const products = await Product.aggregate(pipeline); // Get total count for pagination const countPipeline = pipeline.filter(stage => !stage.$skip && !stage.$limit); countPipeline.push({ $count: 'total' }); const totalResult = await Product.aggregate(countPipeline); const total = totalResult[0]?.total || 0; res.json({ success: true, results: products.length, pagination: { currentPage: page, totalPages: Math.ceil(total / limitNum), totalItems: total }, data: products }); };
