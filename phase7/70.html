// src/models/User.js\nlet users = []; let nextUserId = 1;\nclass User {\n static findAll() { return [...users]; }\n static findById(id) { return users.find(u => u.id === id); }\n static findByEmail(email) { return users.find(u => u.email === email); }\n static create(data) { const newUser = { id: nextUserId++, ...data, createdAt: new Date() }; users.push(newUser); return newUser; }\n static update(id, data) { const index = users.findIndex(u => u.id === id); if(index===-1) return null; users[index] = { ...users[index], ...data, id }; return users[index]; }\n static delete(id) { const index = users.findIndex(u => u.id === id); if(index===-1) return false; users.splice(index,1); return true; }\n}\n\n// src/models/Post.js\nlet posts = []; let nextPostId = 1;\nclass Post {\n static findAll() { return [...posts]; }\n static findById(id) { return posts.find(p => p.id === id); }\n static findByAuthor(authorId) { return posts.filter(p => p.authorId === authorId); }\n static create(data) { const newPost = { id: nextPostId++, ...data, views:0, likes:0, createdAt: new Date(), updatedAt: new Date() }; posts.push(newPost); return newPost; }\n static update(id, data) { const index = posts.findIndex(p => p.id === id); if(index===-1) return null; posts[index] = { ...posts[index], ...data, id, updatedAt: new Date() }; return posts[index]; }\n static delete(id) { const index = posts.findIndex(p => p.id === id); if(index===-1) return false; posts.splice(index,1); return true; }\n static incrementViews(id) { const post = Post.findById(id); if(post) post.views++; return post; }\n static incrementLikes(id) { const post = Post.findById(id); if(post) post.likes++; return post; }\n}\n\n// src/models/Comment.js\nlet comments = []; let nextCommentId = 1;\nclass Comment {\n static findAll() { return [...comments]; }\n static findByPost(postId) { return comments.filter(c => c.postId === postId); }\n static create(data) { const newComment = { id: nextCommentId++, ...data, createdAt: new Date() }; comments.push(newComment); return newComment; }\n static delete(id) { const index = comments.findIndex(c => c.id === id); if(index===-1) return false; comments.splice(index,1); return true; }\n}\nmodule.exports = { User, Post, Comment };
📦 Models handle ALL data operations. User manages authentication, Post manages blog entries with views/likes, Comment manages nested comments. Each model is pure JavaScript (will connect to MongoDB later).
Utils & Auth Service
// src/utils/appError.js\nclass AppError extends Error {\n constructor(message, statusCode) {\n super(message);\n this.statusCode = statusCode;\n this.isOperational = true;\n }\n}\nmodule.exports = AppError;\n\n// src/utils/catchAsync.js\nconst catchAsync = (fn) => (req, res, next) => fn(req, res, next).catch(next);\nmodule.exports = catchAsync;\n\n// src/services/authService.js\nconst jwt = require('jsonwebtoken');\nconst { User } = require('../models/User');\nconst AppError = require('../utils/appError');\n\nconst generateToken = (userId, role) => jwt.sign({ id: userId, role }, process.env.JWT_SECRET, { expiresIn: '7d' });\n\nconst register = async (name, email, password, role = 'user') => {\n if(User.findByEmail(email)) throw new AppError('Email already exists', 400);\n const newUser = User.create({ name, email, password, role });\n const token = generateToken(newUser.id, newUser.role);\n return { user: { id: newUser.id, name, email, role }, token };\n};\n\nconst login = async (email, password) => {\n const user = User.findByEmail(email);\n if(!user || user.password !== password) throw new AppError('Invalid email or password', 401);\n const token = generateToken(user.id, user.role);\n return { user: { id: user.id, name: user.name, email: user.email, role: user.role }, token };\n};\nmodule.exports = { register, login };
🔐 Register → creates user + JWT token. Login → validates password → returns token. Client includes token in Authorization: Bearer for protected routes.
Middleware (Auth & Validation)
// src/middleware/authMiddleware.js\nconst jwt = require('jsonwebtoken');\nconst catchAsync = require('../utils/catchAsync');\nconst AppError = require('../utils/appError');\nconst { User } = require('../models/User');\n\nconst protect = catchAsync(async (req, res, next) => {\n let token = req.headers.authorization?.split(' ')[1];\n if(!token) return next(new AppError('Not logged in', 401));\n const decoded = jwt.verify(token, process.env.JWT_SECRET);\n const user = User.findById(decoded.id);\n if(!user) return next(new AppError('User no longer exists', 401));\n req.user = user;\n next();\n});\n\nconst restrictTo = (...roles) => (req, res, next) => {\n if(!roles.includes(req.user.role)) return next(new AppError('No permission', 403));\n next();\n};\n\n// src/middleware/validationMiddleware.js\nconst AppError = require('../utils/appError');\nconst validatePost = (req, res, next) => {\n if(!req.body.title || req.body.title.length < 3) return next(new AppError('Title min 3 chars', 400));\n if(!req.body.content || req.body.content.length < 10) return next(new AppError('Content min 10 chars', 400));\n next();\n};\nmodule.exports = { protect, restrictTo, validatePost };
🛡️ protect() extracts JWT from header, verifies it, attaches user to req.user. restrictTo('admin') checks role. validatePost() ensures title/content length before reaching controller.

Server Entry Point (server.js)
// server.js\nrequire('dotenv').config();\nconst express = require('express');\nconst morgan = require('morgan');\nconst cors = require('cors');\nconst helmet = require('helmet');\nconst routes = require('./src/routes');\nconst errorHandler = require('./src/middleware/errorMiddleware');\nconst AppError = require('./src/utils/appError');\n\nconst app = express();\nconst PORT = process.env.PORT || 5000;\n\napp.use(helmet());\napp.use(cors());\napp.use(morgan('dev'));\napp.use(express.json());\napp.use('/api', routes);\napp.all('*', (req,res,next) => next(new AppError(`Can't find ${req.originalUrl}`, 404)));\napp.use(errorHandler);\n\napp.listen(PORT, () => console.log(`✅ Blog API running on http://localhost:${PORT}`));\nprocess.on('unhandledRejection', (err) => { console.error(err); process.exit(1); });\nmodule.exports = app;
