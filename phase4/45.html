3️⃣ Step 1 — Cart Context (Global State with LocalStorage)
// context/CartContext.jsx
import { createContext, useContext, useState, useEffect } from 'react';

const CartContext = createContext();

export function CartProvider({ children }) {
const [cart, setCart] = useState(() => {
const saved = localStorage.getItem('cart');
return saved ? JSON.parse(saved) : [];
});
useEffect(() => localStorage.setItem('cart', JSON.stringify(cart)), [cart]);

const addToCart = (product) => {
setCart(prev => {
const existing = prev.find(item => item.id === product.id);
if (existing) return prev.map(item => item.id === product.id ? {...item, quantity: item.quantity + 1} : item);
return [...prev, { ...product, quantity: 1 }];
});
};
const removeFromCart = (id) => setCart(prev => prev.filter(item => item.id !== id));
const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
const totalPrice = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
return (<CartContext.Provider value={{ cart, addToCart, removeFromCart, totalItems, totalPrice }}>{children}</CartContext.Provider>);
}
export const useCart = () => useContext(CartContext);

// services/api.js
const API_URL = 'https://fakestoreapi.com/products';
export const fetchProducts = async () => {
const response = await fetch(API_URL);
if (!response.ok) throw new Error('Failed to fetch products');
return response.json();
};

5️⃣ Step 3 — Products Page (Listing Products)
function Products() {
const [products, setProducts] = useState([]);
const [loading, setLoading] = useState(true);
const { addToCart } = useCart();
useEffect(() => {
fetchProducts().then(setProducts).catch(setError).finally(() => setLoading(false));
}, []);
if (loading) return <div>Loading...</div>;
return products.map(p => <ProductCard key={p.id} product={p} onAdd={addToCart} />);
}

6️⃣ Step 4 — Cart Page (Display & Manage Cart)
function Cart() {
const { cart, totalItems, totalPrice, removeFromCart } = useCart();
if (cart.length === 0) return <div>Empty cart</div>;
return cart.map(item => <CartItem key={item.id} item={item} onRemove={removeFromCart} />);
}

⚠️ Common Mistakes to Avoid
  • ❌ Forgetting to wrap app with <CartProvider> → useCart returns undefined
  • ❌ Not handling loading/error states for API calls → app may crash
  • ❌ Mutating cart state directly (always use setCart with new array)
  • ❌ Not parsing localStorage data with JSON.parse → storing as string
  • ❌ Using <a href> instead of <Link to> → causes page refresh
💼 Common Interview Questions

❓ How would you persist cart data?

Use localStorage with useEffect to sync cart changes, so data survives page refresh.

❓ Why Context API for cart?

Cart data is needed by Navbar, Products, Cart pages — Context avoids prop drilling.

❓ How to handle API loading states?

Use loading state variable, show loading indicator while fetching, hide when done.
