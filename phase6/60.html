Step 3: Create Custom Modules (logger, utils, routes)
📝 modules/logger.js – Event-based Logger
// modules/logger.js const EventEmitter = require('events'); const fs = require('fs').promises; const path = require('path'); class Logger extends EventEmitter { constructor() { super(); this.logFilePath = path.join(__dirname, '..', 'logs', 'requests.log'); this.setupLogFile(); } async setupLogFile() { const logDir = path.join(__dirname, '..', 'logs'); try { await fs.mkdir(logDir, { recursive: true }); } catch (err) { console.error('Error creating logs folder:', err.message); } } async log(message, type = 'INFO') { const timestamp = new Date().toISOString(); const logEntry = `[${timestamp}] [${type}] ${message}\n`; console.log(logEntry.trim()); try { await fs.appendFile(this.logFilePath, logEntry); } catch (err) { console.error('Failed to write log:', err.message); } this.emit('logged', { message, type, timestamp }); } } module.exports = new Logger();
📝 modules/utils.js – Helper Functions
// modules/utils.js const fs = require('fs').promises; const path = require('path'); async function serveStaticFile(filePath, contentType) { try { const fullPath = path.join(__dirname, '..', 'public', filePath); const data = await fs.readFile(fullPath); return { success: true, data, contentType }; } catch (error) { return { success: false, error: error.message }; } } function getContentType(filePath) { const ext = path.extname(filePath).toLowerCase(); const types = { '.html': 'text/html', '.css': 'text/css', '.js': 'application/javascript', '.json': 'application/json', '.png': 'image/png', '.jpg': 'image/jpeg', '.txt': 'text/plain' }; return types[ext] || 'text/plain'; } module.exports = { serveStaticFile, getContentType };
