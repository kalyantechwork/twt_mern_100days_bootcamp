1. What is a REST API?

REST (Representational State Transfer) is an architectural style for building web APIs. REST APIs use HTTP methods to perform operations on resources.

REST API Principles:
• Resources are identified by URLs (e.g., /users, /products/5)
• HTTP methods specify action: GET (read), POST (create), PUT (update), DELETE (remove)
• Stateless: Each request contains all necessary information
• Responses use proper HTTP status codes
📋 HTTP Method → Action

GET → Read data
POST → Create data
PUT → Update entire resource
PATCH → Partial update
DELETE → Remove data

📊 HTTP Status Codes

200 OK → Success
201 Created → Resource created
400 Bad Request → Client error
401 Unauthorized → Auth required
404 Not Found → Resource missing
500 Server Error → Server problem

📁 src/controllers/productController.js
const asyncHandler = require('../utils/asyncHandler'); const { Product } = require('../models/Product'); // @desc Get all products with filtering, sorting, pagination // @route GET /api/v1/products // @access Public const getAllProducts = asyncHandler(async (req, res) => { let products = Product.findAll(); // ========== FILTERING ========== // Support filtering by category, inStock, minPrice, maxPrice const { category, inStock, minPrice, maxPrice, search } = req.query; if (category) { products = products.filter(p => p.category.toLowerCase() === category.toLowerCase()); } if (inStock !== undefined) { const isInStock = inStock === 'true'; products = products.filter(p => p.inStock === isInStock); } if (minPrice) { products = products.filter(p => p.price >= parseFloat(minPrice)); } if (maxPrice) { products = products.filter(p => p.price <= parseFloat(maxPrice)); } if (search) { products = products.filter(p => p.name.toLowerCase().includes(search.toLowerCase())); } // ========== SORTING ========== const sortBy = req.query.sortBy || 'id'; const sortOrder = req.query.sortOrder === 'desc' ? -1 : 1; products.sort((a, b) => { if (a[sortBy] < b[sortBy]) return -1 * sortOrder; if (a[sortBy] > b[sortBy]) return 1 * sortOrder; return 0; }); // ========== PAGINATION ========== const page = parseInt(req.query.page) || 1; const limit = parseInt(req.query.limit) || 10; const startIndex = (page - 1) * limit; const endIndex = startIndex + limit; const paginatedProducts = products.slice(startIndex, endIndex); // Pagination metadata const pagination = {}; if (endIndex < products.length) { pagination.next = { page: page + 1, limit }; } if (startIndex > 0) { pagination.prev = { page: page - 1, limit }; } res.status(200).json({ success: true, count: products.length, pagination, data: paginatedProducts }); }); // @desc Get single product // @route GET /api/v1/products/:id // @access Public const getProductById = asyncHandler(async (req, res) => { const id = parseInt(req.params.id); const product = Product.findById(id); if (!product) { return res.status(404).json({ success: false, message: `Product with id ${id} not found` }); } res.status(200).json({ success: true, data: product }); }); // @desc Create new product // @route POST /api/v1/products // @access Public const createProduct = asyncHandler(async (req, res) => { const { name, price, category, inStock } = req.body; // Validation const errors = []; if (!name || name.length < 2) { errors.push('Product name must be at least 2 characters'); } if (!price || price <= 0) { errors.push('Product price must be a positive number'); } if (!category) { errors.push('Product category is required'); } if (errors.length > 0) { return res.status(400).json({ success: false, errors }); } const newProduct = Product.create({ name, price: parseFloat(price), category, inStock: inStock !== undefined ? inStock : true }); res.status(201).json({ success: true, data: newProduct }); }); // @desc Update product // @route PUT /api/v1/products/:id // @access Public const updateProduct = asyncHandler(async (req, res) => { const id = parseInt(req.params.id); const { name, price, category, inStock } = req.body; const existingProduct = Product.findById(id); if (!existingProduct) { return res.status(404).json({ success: false, message: `Product with id ${id} not found` }); } const updatedProduct = Product.update(id, { name: name || existingProduct.name, price: price !== undefined ? parseFloat(price) : existingProduct.price, category: category || existingProduct.category, inStock: inStock !== undefined ? inStock : existingProduct.inStock }); res.status(200).json({ success: true, data: updatedProduct }); }); // @desc Delete product // @route DELETE /api/v1/products/:id // @access Public const deleteProduct = asyncHandler(async (req, res) => { const id = parseInt(req.params.id); const product = Product.findById(id); if (!product) { return res.status(404).json({ success: false, message: `Product with id ${id} not found` }); } Product.delete(id); res.status(200).json({ success: true, message: `Product ${id} deleted successfully`, data: product }); }); // @desc Get product statistics // @route GET /api/v1/products/stats/summary // @access Public const getProductStats = asyncHandler(async (req, res) => { const products = Product.findAll(); const stats = { totalProducts: products.length, totalValue: products.reduce((sum, p) => sum + p.price, 0), averagePrice: products.reduce((sum, p) => sum + p.price, 0) / products.length, categories: [...new Set(products.map(p => p.category))], inStockCount: products.filter(p => p.inStock).length, outOfStockCount: products.filter(p => !p.inStock).length }; res.status(200).json({ success: true, data: stats }); }); module.exports = { getAllProducts, getProductById, createProduct, updateProduct, deleteProduct, getProductStats };

6. REST API Best Practices Checklist
✅ Use Proper HTTP Methods

GET (read), POST (create), PUT (update), DELETE (remove)

✅ Use Correct Status Codes

200 OK, 201 Created, 400 Bad Request, 404 Not Found, 500 Error

✅ Use Plural Nouns for Resources

/users (not /user), /products (not /product)

✅ Version Your API

/api/v1/products, /api/v2/products

✅ Use Consistent Response Format

{ success, data, message, errors }

✅ Implement Filtering, Sorting, Pagination

?category=electronics&sortBy=price&page=2&limit=10
