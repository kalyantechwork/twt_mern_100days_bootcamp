1. What is Express.js?

Express.js is a fast, unopinionated, and minimalist web framework for Node.js. It provides powerful features to build web applications and REST APIs with less code.

+ = Powerful Backend in Minutes!
Express sits on top of Node.js and simplifies everything!

❌ Without Express (Raw Node.js)

Manual routing, manual body parsing, complex request handling

const http = require('http'); const server = http.createServer((req,res)=>{ if(req.url === '/'){...} else if(req.url === '/users'){...} });

✅ With Express.js

Clean routes, automatic body parsing, middleware support

const express = require('express'); const app = express(); app.get('/', (req,res)=>{...}); app.get('/users', (req,res)=>{...});

5. Understanding Express Basics
📝 app.get(), app.post(), app.put(), app.delete()
const express = require('express'); const app = express(); // GET - Read data app.get('/users', (req, res) => { res.json({ message: 'Get all users' }); }); // POST - Create new data app.post('/users', (req, res) => { res.json({ message: 'Create a new user' }); }); // PUT - Update entire data app.put('/users/:id', (req, res) => { res.json({ message: `Update user ${req.params.id}` }); }); // DELETE - Remove data app.delete('/users/:id', (req, res) => { res.json({ message: `Delete user ${req.params.id}` }); }); app.listen(3000);
📝 Route Parameters (Dynamic URLs)
// :id is a dynamic parameter app.get('/users/:id', (req, res) => { const userId = req.params.id; res.send(`User ID: ${userId}`); }); // Multiple parameters app.get('/products/:category/:productId', (req, res) => { const { category, productId } = req.params; res.send(`Category: ${category}, Product: ${productId}`); }); // Example: /users/123 → Output: "User ID: 123" // Example: /products/electronics/456 → Output: "Category: electronics, Product: 456"
📝 Query Parameters (URL ?key=value)
// Query parameters: /search?q=express&page=2 app.get('/search', (req, res) => { const query = req.query.q; const page = req.query.page || 1; res.send(`Searching for "${query}" on page ${page}`); }); // Example: /search?q=nodejs&page=3 // Output: "Searching for 'nodejs' on page 3"

7. Serving Static Files (HTML, CSS, Images)

Express makes it super easy to serve static files like HTML, CSS, and images:

// Create a 'public' folder with your static files // public/index.html, public/style.css, public/logo.png const express = require('express'); const app = express(); // Serve all files from 'public' folder app.use(express.static('public')); // Now any file in public folder is accessible: // http://localhost:3000/index.html // http://localhost:3000/style.css // http://localhost:3000/logo.png app.listen(3000);
Pro Tip: express.static() is a middleware – functions that run between request and response. We'll learn more about middleware tomorrow!
8. Parsing Request Body (Getting Data from POST)

To read data sent in POST requests (JSON or form data), Express provides built-in middleware:

const express = require('express'); const app = express(); // Built-in middleware to parse JSON data app.use(express.json()); // Built-in middleware to parse form data (URL-encoded) app.use(express.urlencoded({ extended: true })); // POST endpoint to receive JSON app.post('/api/users', (req, res) => { const { name, email } = req.body; // Access parsed data console.log('Received:', name, email); res.json({ message: `User ${name} created!` }); }); // Test with curl: // curl -X POST http://localhost:3000/api/users \ // -H "Content-Type: application/json" \ // -d '{"name":"Ramesh","email":"ramesh@example.com"}' app.listen(3000);
9. Express vs Raw Node.js – Why Express Wins
Raw Node.js HTTP Module
const http = require('http'); const server = http.createServer((req,res)=>{ const url = req.url; if(url === '/' && req.method === 'GET'){ res.writeHead(200); res.end('Home'); } else if(url === '/users' && req.method === 'GET'){ let body = ''; req.on('data', chunk => body+=chunk); req.on('end',()=>{ const data = JSON.parse(body); res.writeHead(200); res.end(JSON.stringify(data)); }); } });

❌ Manual routing
❌ Manual body parsing
❌ Complex code
❌ More bugs

Express.js
const express = require('express'); const app = express(); app.use(express.json()); app.get('/', (req,res)=>{ res.send('Home'); }); app.post('/users', (req,res)=>{ const data = req.body; res.json(data); }); app.listen(3000);

✅ Clean routing
✅ Auto body parsing
✅ Less code
✅ More readable
