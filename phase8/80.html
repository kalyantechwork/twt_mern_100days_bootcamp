// models/User.js const mongoose = require('mongoose'); const bcrypt = require('bcryptjs'); const userSchema = new mongoose.Schema({ name: { type: String, required: [true, 'Name is required'], trim: true, minlength: [2, 'Name must be at least 2 characters'], maxlength: [50, 'Name cannot exceed 50 characters'] }, email: { type: String, required: [true, 'Email is required'], unique: true, lowercase: true, trim: true, match: [/^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/, 'Please enter a valid email'] }, password: { type: String, required: [true, 'Password is required'], minlength: [6, 'Password must be at least 6 characters'], select: false }, role: { type: String, enum: ['student', 'instructor', 'admin'], default: 'student' }, profile: { avatar: String, bio: { type: String, maxlength: 500 }, occupation: String, institution: String }, enrolledCourses: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Course' }], completedCourses: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Course' }], wishlist: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Course' }], isActive: { type: Boolean, default: true }, lastLogin: Date }, { timestamps: true }); // Indexes userSchema.index({ email: 1 }); userSchema.index({ role: 1 }); userSchema.index({ 'profile.institution': 1 }); // Hash password before saving userSchema.pre('save', async function(next) { if (!this.isModified('password')) return next(); this.password = await bcrypt.hash(this.password, 12); next(); }); // Instance method to check password userSchema.methods.comparePassword = async function(candidatePassword) { return await bcrypt.compare(candidatePassword, this.password); }; // Virtual for full profile name userSchema.virtual('fullProfile').get(function() { return { name: this.name, email: this.email, role: this.role, bio: this.profile?.bio, avatar: this.profile?.avatar }; }); const User = mongoose.model('User', userSchema); module.exports = User;

// Indexes for performance courseSchema.index({ title: 'text', description: 'text', tags: 'text' }); courseSchema.index({ category: 1, level: 1 }); courseSchema.index({ price: 1 }); courseSchema.index({ rating: -1 }); courseSchema.index({ studentsEnrolled: -1 }); // Pre-save middleware to generate slug courseSchema.pre('save', function(next) { if (this.isModified('title')) { this.slug = this.title.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, ''); } next(); }); // Virtual for modules courseSchema.virtual('modules', { ref: 'Module', localField: '_id', foreignField: 'course' }); // Virtual for reviews courseSchema.virtual('reviews', { ref: 'Review', localField: '_id', foreignField: 'course' }); // Virtual for enrollment count courseSchema.virtual('enrollmentCount').get(function() { return this.studentsEnrolled; }); // Virtual for isFree courseSchema.virtual('isFree').get(function() { return this.price === 0; }); // Method to update rating courseSchema.methods.updateRating = async function() { const Review = mongoose.model('Review'); const result = await Review.aggregate([ { $match: { course: this._id } }, { $group: { _id: null, avgRating: { $avg: '$rating' }, count: { $sum: 1 } } } ]); if (result.length > 0) { this.rating = Math.round(result[0].avgRating * 10) / 10; this.ratingCount = result[0].count; await this.save(); } }; const Course = mongoose.model('Course', courseSchema); module.exports = Course;

// models/Enrollment.js const mongoose = require('mongoose'); const enrollmentSchema = new mongoose.Schema({ user: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true }, course: { type: mongoose.Schema.Types.ObjectId, ref: 'Course', required: true }, progress: { type: Number, default: 0, min: 0, max: 100 }, completedLessons: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Lesson' }], lastAccessed: Date, completedAt: Date, certificateIssued: { type: Boolean, default: false }, status: { type: String, enum: ['active', 'completed', 'dropped'], default: 'active' } }, { timestamps: true }); // Unique compound index to prevent duplicate enrollments enrollmentSchema.index({ user: 1, course: 1 }, { unique: true }); enrollmentSchema.index({ user: 1, status: 1 }); enrollmentSchema.index({ course: 1, progress: 1 }); // Method to update progress enrollmentSchema.methods.updateProgress = async function() { const Lesson = mongoose.model('Lesson'); const Module = mongoose.model('Module'); const modules = await Module.find({ course: this.course }); let totalLessons = 0; for (const module of modules) { const lessons = await Lesson.find({ module: module._id }); totalLessons += lessons.length; } const progress = (this.completedLessons.length / totalLessons) * 100; this.progress = Math.round(progress); if (this.progress === 100 && this.status !== 'completed') { this.status = 'completed'; this.completedAt = new Date(); } await this.save(); }; const Enrollment = mongoose.model('Enrollment', enrollmentSchema); module.exports = Enrollment;

// models/Review.js const mongoose = require('mongoose'); const reviewSchema = new mongoose.Schema({ user: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true }, course: { type: mongoose.Schema.Types.ObjectId, ref: 'Course', required: true }, rating: { type: Number, required: true, min: 1, max: 5 }, comment: { type: String, required: true, maxlength: 500 }, isApproved: { type: Boolean, default: true } }, { timestamps: true }); // Prevent duplicate reviews from same user for same course reviewSchema.index({ user: 1, course: 1 }, { unique: true }); reviewSchema.index({ course: 1, rating: -1 }); // After saving, update course rating reviewSchema.post('save', async function() { const Course = mongoose.model('Course'); const course = await Course.findById(this.course); if (course) { await course.updateRating(); } }); reviewSchema.post('remove', async function() { const Course = mongoose.model('Course'); const course = await Course.findById(this.course); if (course) { await course.updateRating(); } }); const Review = mongoose.model('Review', reviewSchema); module.exports = Review;
API Features Utility (Pagination, Filtering, Sorting)
// utils/apiFeatures.js class APIFeatures { constructor(query, queryString) { this.query = query; this.queryString = queryString; } filter() { const queryObj = { ...this.queryString }; const excludedFields = ['page', 'sort', 'limit', 'fields', 'search']; excludedFields.forEach(el => delete queryObj[el]); let queryStr = JSON.stringify(queryObj); queryStr = queryStr.replace(/\b(gt|gte|lt|lte|in)\b/g, match => `$${match}`); this.query = this.query.find(JSON.parse(queryStr)); return this; } search() { if (this.queryString.search) { this.query = this.query.find({ $or: [ { title: { $regex: this.queryString.search, $options: 'i' } }, { description: { $regex: this.queryString.search, $options: 'i' } }, { tags: { $in: [new RegExp(this.queryString.search, 'i')] } } ] }); } return this; } sort() { if (this.queryString.sort) { const sortBy = this.queryString.sort.split(',').join(' '); this.query = this.query.sort(sortBy); } else { this.query = this.query.sort('-createdAt'); } return this; } limitFields() { if (this.queryString.fields) { const fields = this.queryString.fields.split(',').join(' '); this.query = this.query.select(fields); } else { this.query = this.query.select('-__v'); } return this; } paginate() { const page = parseInt(this.queryString.page) || 1; const limit = parseInt(this.queryString.limit) || 10; const skip = (page - 1) * limit; this.query = this.query.skip(skip).limit(limit); this.paginationInfo = { page, limit, skip }; return this; } async getPaginationMetadata(totalDocuments) { const { page, limit } = this.paginationInfo; const totalPages = Math.ceil(totalDocuments / limit); return { currentPage: page, totalPages, totalDocuments, documentsPerPage: limit, hasNextPage: page < totalPages, hasPrevPage: page > 1, nextPage: page < totalPages ? page + 1 : null, prevPage: page > 1 ? page - 1 : null }; } } module.exports = APIFeatures;
Sample Routes with Advanced Queries
// routes/courseRoutes.js const express = require('express'); const router = express.Router(); const Course = require('../models/Course'); const Enrollment = require('../models/Enrollment'); const APIFeatures = require('../utils/apiFeatures'); const catchAsync = require('../utils/catchAsync'); // GET /api/courses - Get all courses with advanced filtering router.get('/courses', catchAsync(async (req, res) => { const features = new APIFeatures(Course.find({ isPublished: true }), req.query) .filter() .search() .sort() .limitFields() .paginate(); const courses = await features.query .populate('instructor', 'name profile.avatar') .populate('category', 'name'); const total = await Course.countDocuments({ isPublished: true }); const pagination = await features.getPaginationMetadata(total); res.json({ success: true, results: courses.length, pagination, data: courses }); })); // GET /api/courses/:id - Get course with full details router.get('/courses/:id', catchAsync(async (req, res) => { const course = await Course.findById(req.params.id) .populate('instructor', 'name email profile.bio profile.avatar') .populate('category', 'name') .populate({ path: 'modules', populate: { path: 'lessons', select: 'title duration order isFree' } }) .populate({ path: 'reviews', populate: { path: 'user', select: 'name profile.avatar' }, options: { limit: 5, sort: { createdAt: -1 } } }); if (!course) { return res.status(404).json({ success: false, message: 'Course not found' }); } res.json({ success: true, data: course }); })); // GET /api/courses/:id/students - Get enrolled students with progress router.get('/courses/:id/students', catchAsync(async (req, res) => { const { page = 1, limit = 10, sortBy = 'progress', sortOrder = 'desc' } = req.query; const enrollments = await Enrollment.find({ course: req.params.id }) .populate('user', 'name email profile.avatar') .sort({ [sortBy]: sortOrder === 'desc' ? -1 : 1 }) .skip((page - 1) * limit) .limit(parseInt(limit)); const total = await Enrollment.countDocuments({ course: req.params.id }); res.json({ success: true, results: enrollments.length, pagination: { currentPage: parseInt(page), totalPages: Math.ceil(total / limit), totalStudents: total }, data: enrollments }); })); // GET /api/courses/stats/popular - Most popular courses (aggregation) router.get('/courses/stats/popular', catchAsync(async (req, res) => { const popularCourses = await Course.aggregate([ { $match: { isPublished: true } }, { $lookup: { from: 'enrollments', localField: '_id', foreignField: 'course', as: 'enrollments' } }, { $addFields: { enrollmentCount: { $size: '$enrollments' } } }, { $sort: { enrollmentCount: -1, rating: -1 } }, { $limit: 10 }, { $project: { title: 1, thumbnail: 1, price: 1, rating: 1, enrollmentCount: 1 } } ]); res.json({ success: true, data: popularCourses }); })); // GET /api/courses/stats/by-category - Course count by category router.get('/courses/stats/by-category', catchAsync(async (req, res) => { const stats = await Course.aggregate([ { $match: { isPublished: true } }, { $group: { _id: '$category', count: { $sum: 1 }, avgPrice: { $avg: '$price' }, avgRating: { $avg: '$rating' }, totalStudents: { $sum: '$studentsEnrolled' } } }, { $lookup: { from: 'categories', localField: '_id', foreignField: '_id', as: 'categoryInfo' } }, { $sort: { count: -1 } } ]); res.json({ success: true, data: stats }); })); module.exports = router;
